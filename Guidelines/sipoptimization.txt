function getInvestmentBudget() {
  // Load Step 1 data and investment rule
  const data = JSON.parse(localStorage.getItem('we_step1') || '{}');
  const investRule = getInvestRule();
  
  // Calculate monthly income
  const monthlyIncome = (data.income || []).reduce((sum, item) => sum + (item.value || 0), 0);
  
  // Calculate monthly expenses
  const monthlyExpenses = (data.expenses || []).reduce((sum, item) => {
    return sum + (item.monthly || 0) + ((item.annual || 0) / 12);
  }, 0);
  
  // Calculate existing EMIs from liabilities
  const existingEMIs = (data.assets || []).reduce((sum, item) => {
    return sum + ((item.type && item.type.toLowerCase() === 'liability') ? (item.emi || 0) : 0);
  }, 0);
  
  // Budget = Income × Investment Percentage (default 30%)
  const pct = investRule.pct || 30;
  return (monthlyIncome * pct) / 100;
}

3. rowSipFor() - Lines 238-274
Purpose: Calculate SIP needed for a goal using binary search


function rowSipFor(futureValue, yearsLeft, lumpsum, customStepUp = null, isEmergencyFund = false) {
  // Get expected annual return based on horizon
  const annR = preRetAnnualReturn(yearsLeft, isEmergencyFund);
  const i = annR / 12;                    // Monthly rate
  const n = Math.max(1, yearsLeft * 12);  // Total months
  
  // Get step-up rate (default from investRule, or custom)
  const stepUp = customStepUp !== null 
    ? customStepUp 
    : Math.min(0.10, Math.max(0, parseFloat(investRule.autoStepUp || '0') / 100));
  
  // Calculate Future Value of lumpsum investment
  const fvLump = Math.max(0, (lumpsum || 0)) * Math.pow(1 + i, n);
  
  // Amount needed from SIP (after lumpsum contribution)
  const need = Math.max(0, futureValue - fvLump);
  
  // If lumpsum is sufficient, no SIP needed
  if (need <= 0) return 0;
  
  // Inner function: Calculate FV from a given initial SIP
  function fvFromSip(initialSip) {
    if (n <= 0) return 0;
    
    let totalFV = 0;
    const totalYears = Math.ceil(n / 12);
    
    for (let year = 0; year < totalYears; year++) {
      const monthsInYear = Math.min(12, n - year * 12);
      // SIP increases by step-up % each year
      const sipForThisYear = initialSip * Math.pow(1 + stepUp, year);
      
      // Compound each month's contribution
      for (let month = 0; month < monthsInYear; month++) {
        const monthsRemaining = n - (year * 12 + month);
        const futureValue = sipForThisYear * Math.pow(1 + i, monthsRemaining);
        totalFV += futureValue;
      }
    }
    return totalFV;
  }
  
  // BINARY SEARCH to find initial SIP
  let lo = 0;
  let hi = (need > 0 ? need / n * 20 + 10000 : 0);
  
  for (let k = 0; k < 100; k++) {
    const mid = (lo + hi) / 2;
    const got = fvFromSip(mid);
    
    // If within ₹1000 tolerance, return this SIP
    if (Math.abs(got - need) < 1000) return mid;
    
    if (got >= need) hi = mid;
    else lo = mid;
  }
  
  return hi;
}
Algorithm Explanation:

Calculate monthly return rate from annual return
Calculate how much lumpsum will grow to
Calculate remaining amount needed from SIP
Use binary search (100 iterations max) to find initial SIP
Each iteration calculates FV with step-up applied annually
Stop when within ₹1000 of target

function optimizePlan() {
  // Prevent concurrent optimization
  if (window.optimizationInProgress) {
    console.log('Optimization already in progress, skipping...');
    return;
  }

  // Check if max attempts reached
  if (window.optimizationCount >= window.maxOptimizations) {
    alert('Maximum optimization attempts reached (3). Please use the RESET button to start over.');
    return;
  }

  // Mark as in progress and increment counter
  window.optimizationInProgress = true;
  window.optimizationCount++;

  // Get budget
  const budget = getInvestmentBudget();
  if (budget <= 0) {
    alert('No investment budget found. Please complete the previous steps first.');
    window.optimizationInProgress = false;
    return;
  }

  // Collect all goals with their data
  const goalsData = [];
  Array.from(tbody.querySelectorAll('tr')).forEach(tr => {
    const name = tr.querySelector('td').textContent;
    const fv = parseFloat(tr.querySelector('input[data-fv]').value) || 0;
    const lump = parseFloat(tr.querySelector('input[data-lump]').value) || 0;
    const targetYear = parseInt(tr.querySelector('input[data-year]').value) || nowYear;
    const yLeft = Math.max(0, targetYear - nowYear);
    const isEmergencyFund = name.toLowerCase().includes('emergency');
    const sip = rowSipFor(fv, yLeft, lump, null, isEmergencyFund);
    
    // Get priority (High for retirement/emergency, else from stored goals)
    let priority = 'Medium';
    if (name.toLowerCase().includes('retire') || name.toLowerCase().includes('emergency')) {
      priority = 'High';
    } else {
      const storedGoal = goals.find(g => g.name === name);
      if (storedGoal) priority = storedGoal.priority;
    }
    
    goalsData.push({ tr, name, fv, lump, yLeft, targetYear, sip, priority, fvInput, yearInput, isEmergencyFund });
  });

  // Calculate current total SIP
  let currentTotal = goalsData.reduce((sum, g) => sum + g.sip, 0);

  // If already within budget, no optimization needed
  if (currentTotal <= budget) {
    alert('Your current plan already fits within your budget!');
    window.optimizationInProgress = false;
    return;
  }

  // Calculate shortfall
  const shortfall = currentTotal - budget;

  // Separate goals by priority
  const highPriorityGoals = goalsData.filter(g => g.priority === 'High');
  const mediumPriorityGoals = goalsData.filter(g => g.priority === 'Medium');
  const lowPriorityGoals = goalsData.filter(g => g.priority === 'Low');
}

Why Step-Up Helps:

Higher annual step-up = lower initial SIP needed
